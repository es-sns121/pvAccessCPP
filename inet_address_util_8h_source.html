34 /** interface description
35  * ifaceAddr - Local address of interface. Will never be AF_UNSPEC
36  * ifaceBCast - Broadcast address of interface.
37  * AF_UNSPEC for loopback and point to point (eg. some VPNs)
38  * !WIN32 bind this to receive bcasts.
39  * ifaceDest - Destination/Peer for point to point.
40  * Copy of ifaceBCast for bcast.
41  * AF_UNSPEC for loopback.
42  * send UDP searches to this address
43  */

112 struct comp_osiSock_lt {
113  bool operator()(const osiSockAddr& a, const osiSockAddr& b) const {
114  if(a.sa.sa_family<b.sa.sa_family) return true;
115  if((a.sa.sa_family==b.sa.sa_family)&&(a.ia.sin_addr.s_addr
116  <b.ia.sin_addr.s_addr)) return true;
117  if((a.sa.sa_family==b.sa.sa_family)&&(a.ia.sin_addr.s_addr
118  ==b.ia.sin_addr.s_addr)&&(a.ia.sin_port
119  <b.ia.sin_port)) return true;
120  return false;
121  }
